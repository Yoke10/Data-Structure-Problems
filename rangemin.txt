#include<iostream>
using namespace std;
void find(int arr[],int n,int a,int b){
    int min=arr[a];
    for(int i=a;i<=b;i++){
        if(arr[i]<min){
            min=arr[i];
        }
    }
    cout<<"Minimum of ["<<a<<", "<<b<<"] "<<"is "<<min<<endl;
}
int main(){
    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    int m,a,b;
    cin>>m;
    for(int i=0;i<m;i++){
        cin>>a>>b;
        find(arr,n,a,b);
    }
}
